SimplexVertex class 

class SimplexVertex {
  final Vector wA; // support point in shapeA
  final Vector wB; // support point in shapeB
  final Vector w; // wB - wA
  num a; // barycentric coordinate for closest point
  int indexA; // wA index
  int indexB; // wB index

  SimplexVertex() :
    wA = new Vector(),
    wB = new Vector(),
    w = new Vector(),
    a = 0,
    indexA = 0,
    indexB = 0 { }

  void setFrom(SimplexVertex sv) {
    wA.setFrom(sv.wA);
    wB.setFrom(sv.wB);
    w.setFrom(sv.w);
    a = sv.a;
    indexA = sv.indexA;
    indexB = sv.indexB;
  }

  String toString() => "wA: $wA, wB: $wB, w: $w";
}
