ClipVertex class

Used for computing contact manifolds.

class ClipVertex {
 Vector v;
 ContactID id;

 ClipVertex() :
   v = new Vector(),
   id = new ContactID() { }

 void setFrom(ClipVertex cv){
   v.setFrom(cv.v);
   id.setFrom(cv.id);
 }
}
